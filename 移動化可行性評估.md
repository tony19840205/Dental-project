```javascript
// sw.js - 快取策略
const CACHE_NAME = 'dental-fhir-v1';
const urlsToCache = [
  '/',
  '/dental-ehr.html',
  '/TestPatients.json',
  '/DentalCrossClinicRules.cql'
];

self.addEventListener('install', event => {
  event.waitUntil(
    caches.open(CACHE_NAME)
      .then(cache => cache.addAll(urlsToCache))
  );
});

// 網路優先，失敗則使用快取
self.addEventListener('fetch', event => {
  event.respondWith(
    fetch(event.request)
      .catch(() => caches.match(event.request))
  );
});
```

### Phase 3：離線功能（2-3 天）

#### 3.1 IndexedDB 整合
```javascript
// 使用 Dexie.js（IndexedDB 包裝庫）
const db = new Dexie('DentalFHIR');
db.version(1).stores({
  patients: 'id, identifier, name',
  procedures: 'id, patientId, date',
  encounters: 'id, patientId, date',
  syncQueue: '++id, action, resource, timestamp'
});

// 離線時將變更加入佇列
async function saveProcedureOffline(procedureData) {
  await db.syncQueue.add({
    action: 'CREATE',
    resource: 'Procedure',
    data: procedureData,
    timestamp: new Date()
  });
}

// 恢復連線時同步
async function syncOfflineData() {
  const queue = await db.syncQueue.toArray();
  for (const item of queue) {
    try {
      await fhirRequest('POST', `/${item.resource}`, item.data);
      await db.syncQueue.delete(item.id);
    } catch (error) {
      console.error('同步失敗', error);
    }
  }
}
```

#### 3.2 連線狀態偵測
```javascript
window.addEventListener('online', () => {
  updateConnectionStatus(true);
  syncOfflineData();
});

window.addEventListener('offline', () => {
  updateConnectionStatus(false);
});

function updateConnectionStatus(isOnline) {
  const indicator = document.querySelector('.status-indicator');
  if (isOnline) {
    indicator.classList.add('green');
    indicator.classList.remove('red');
  } else {
    indicator.classList.add('red');
    indicator.classList.remove('green');
  }
}
```
